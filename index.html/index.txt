<script>
    // ★重要：ここにメモしたGASのURLを貼り付けてください
    const GAS_URL = "https://script.google.com/macros/s/https://script.google.com/macros/s/AKfycbx1xQzdKXhJaYZnwtHVvu_hS64Lr--hYpfWO3PH6dn_kFf2OVOlZPNG6bYTJa_z5Nrr/exec/exec";

    document.getElementById('reservationForm').addEventListener('submit', function(e) {
        e.preventDefault();
        const btn = document.getElementById('submitBtn');
        btn.innerText = "送信中...";
        btn.disabled = true;

        const formData = new FormData(this);
        const data = {};
        formData.forEach((value, key) => { data[key] = value });

        // GASへデータを飛ばす処理
        fetch(GAS_URL, {
            method: "POST",
            body: JSON.stringify(data),
            mode: "no-cors" // これによりブラウザのセキュリティ制限を回避します
        }).then(() => {
            alert("ありがとうございます。送信が完了しました。");
            btn.innerText = "送信完了";
        }).catch(err => {
            alert("エラーが発生しました。");
            btn.disabled = false;
            btn.innerText = "再試行する";
        });
    });
</script>
